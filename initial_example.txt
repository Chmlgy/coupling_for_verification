RANDOM_BIT_COUNT = 2
OBS_SET = ["w1"]

def circuit(secret, randomness):
    r0, d0 = randomness

    w0 = (secret + r0) % 2
    w1 = (w0 * r0) % 2
    w2 = (w1 + d0) % 2
    
    return {
        "w0": w0, "w1": w1, "w2": w2
    }


def get_wire_names():
    return ["w0", "w1", "w2"]
